    📚 核心 API 示例
    // 1. 可观察对象 (Proxy 代理) const state = reactive({ count: 0 }) // 2. 副作用 (依赖自动收集) effect(() => { console.log(state.count) // 收集 count 为依赖 }) // 3. 计算属性 (带缓存) const doubled = computed(() => state.count * 2) // 4. 批量更新 (避免重复运行) batch(() => { state.count++ state.count++ }) // effect 只运行一次
